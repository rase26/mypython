def makewordlist(df):
    #df '' '' ''
    wordlist=pd.Series()
    wordlist=wordlist.append(df['']).append(df['']).append(df[''])
    kw=[]
    #
    for t in wordlist:
        tt=str(t)
        ks=jieba.cut(tt.lower())
        kw.extend(ks)

    #
    rkw= list(filter(lambda a: a not in stopWords and a != '\n', kw))
    return rkw

def wordfreq(data):
    ck=pd.Series() # Series 
    for i in data:
        ck=ck.append(pd.Series(i))

    sc=ck.value_counts() # 
    return sc

def toolwordcloud(data,mask,filename):
    import matplotlib.pyplot as plt
    from wordcloud import WordCloud
    import random
    from palettable.colorbrewer.qualitative import Dark2_8 # 
    
    #
    def color_func(word, font_size, position, orientation, random_state=None, **kwargs):
        return tuple(Dark2_8.colors[random.randint(0,7)]) #random
    
    word_could_dict=Counter(data) #data 
    bg_pic = np.array(Image.open(mask)) #
    
    #
    wordcloud = WordCloud(width = 1000, height = 500,font_path="msjhbd.ttc",max_words=400,mask=bg_pic,background_color="white").generate_from_frequencies(word_could_dict)
    #
    wordcloud.recolor(color_func=color_func, random_state=2) 
    #
    wordcloud.to_file(filename)
    #
    plt.figure(figsize=(15,8))
    plt.imshow(wordcloud)
    plt.axis("off")
    plt.show()
    plt.close()
