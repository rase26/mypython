def jobinfo(res): #爬工作的資料
    #將變數變得外面也可以用
    global job,company,salary,exp,indust,relist,toolist,jcontent,date,jlink

    p=BeautifulSoup(res.text,'html.parser')

    job = pd.Series() #職位 
    company = pd.Series() #公司 
    indust =pd.Series() #產業別 
    toolist=pd.Series() #擅長工具
    relist=pd.Series() #條件要求
    jcontent=pd.Series() #工作內容
    jlink =pd.Series() #職缺網址
    salary = pd.Series() #起薪
    exp= pd.Series() #工作經歷 
    date=pd.Series() #postdate

    print('開始抓了~')

    for i in p.select("article.b-block--top-bord.job-list-item.b-clearfix.js-job-item"): 
        #職位 公司 網址資訊都在這一塊裡面，所以用迴圈一個一個抓
        clink='https:'+ i.select("h2.b-tit a")[0].get('href') #職缺網址
        print(clink)

            company = company.append(pd.Series([i['data-cust-name']])).reset_index(drop=True) #公司
            indust=indust.append(pd.Series([i['data-indcat-desc']])).reset_index(drop=True) #產業別
            
            #工作內容
            jc0=i.select('p.job-list-item__info.b-clearfix.b-content')
            if len(jc0)!=0:
                jcontent=jcontent.append(pd.Series([jc0[0].text])).reset_index(drop=True) 
            else:
                jcontent=jcontent.append(pd.Series(["NA"])).reset_index(drop=True)
            jlink=jlink.append(pd.Series([clink])).reset_index(drop=True) #網址
            
             #薪資 只看起薪
            s=i.select("span.b-tag--default")[0].text[2:]
            if s=='面議': #目前規定面議薪水要4萬以上，故以4萬計算
                s0=40000

                s0=int(s[0].replace(',',''))
            salary = salary.append(pd.Series([s0])).reset_index(drop=True)

            #所需工作經驗
            explist=i.select('ul.b-list-inline.b-clearfix.job-list-intro.b-content li')
            #因為爬出來理面每三個的第二個才是我們需要的,所以用迴圈抓取
            n=1

                e=e.text
                if e != '':
                    if n%3==0:
                        n=1
                    elif n%2== 0:
                        exp=exp.append(pd.Series([e])).reset_index(drop=True)
                        n+=1
                    else:
                        n+=1
            #工作刊登時間
            dl=i.select('span.b-tit__date')[0].text.strip().split("/")
            d="2018-"+dl[0]+"-"+dl[1] #轉成資料庫的時間格式
            date=date.append(pd.Series([d])).reset_index(drop=True)

def importomysql(k,job,company,salary,exp,indust,relist,toolist,jcontent,date,jlink):
    #連接資料庫
    dlink = pymysql.connect(
                host ="localhost",
                user="root",
                passwd="",
                db="job" #資料庫名稱
                )
    #執行資料庫的變數
    cmd = dlink.cursor()

    ##直接加入 jobinfo 中的迴圈
    n=0
    
    #將資料list 裡面資料一個一個取出
    for j in job:
        c=company[n]
        s=str(salary[n])
        e=exp[n]
        i=indust[n]
        r=relist[n]
        t=toolist[n]
        jc=jcontent[n]
        d=date[n]
        l=jlink[n]

        if cmd.execute("select * from `104` where `jobtitle`= %s and `company`=%s",(j,c)) == 0: #查詢資料庫 若資料庫沒有則匯入
            #匯入資料庫的資料
            pm= {
                "keyword":k,
                "jobtitle":j,
                "date":d,
                "company":c,
                "salary":s,
                "experiencetime":e,
                "industry":i,
                "requirement":r,
                "requiretool":t,
                "jobcontent":jc,
                "joblink":l
                }
            
            #執行匯入資料庫 新增列
            cmd.execute("INSERT INTO `104`(`keyword`,`jobtitle`,`date`,`company`,`salary`,`experiencetime`,`industry`,`requirement`,`requiretool`,`jobcontent`,`link`) VALUES(%(keyword)s,%(jobtitle)s,%(date)s,%(company)s,%(salary)s,%(experiencetime)s,%(industry)s,%(requirement)s,%(requiretool)s,%(jobcontent)s,%(joblink)s)",pm)
            dlink.commit #執行
            print('寫入mysql中', n)
            
        elif cmd.execute("select `id`,`keyword` from `104` where `jobtitle`= %s and `company`=%s",(j,c)) !=0: #已有此筆資料，但關鍵字不同，新增關鍵字
            k0=cmd.fetchall()[0] #取出資料已有的關鍵字資訊
            Id=k0[0]     #那筆資料的id
            if k not in k0[1] : #如果沒有新的關鍵字 則新增關鍵字
                k1=k0[1]+' '+k
                cmd.execute("update `104` set `keyword`=%s where `jobtitle`= %s and `company`=%s",(k1,j,c))
                dlink.commit()
                print('mysql中已有此筆資料,在id=%s,新增關鍵字:%s,%s'%(Id,k,n))
            else: #已經有關鍵字了 就不用新增了
                print('mysql中已有此筆資料及關鍵字:%s,在id=%s,%s'%(k,Id,n))
        elif cmd.execute("select `id`,`keyword` from `104` where `jobtitle`= %s and `company`=%s",(j,c)) != 0:
            k0=cmd.fetchall()[0]
            Id=k0[0]
            print('mysql中已有此筆資料,在id=%s,%s'%(Id,n))
        else:
            print('有問題!!')
        n+=1

    dlink.close()
    print('成功寫入mysql')

##畫出數量前幾名的長條圖
def counts(df,column,title,pngname,r):
    #計算提到的次數 存成字典
    b=[]
    count={}
    for tt in df[column] :
        t=tt.split(' ')
        for i in t:
            if i not in b:
                count.update({i:1})
                b.append(i)
            else:
                count[i]+=1
    if '' in count:
        del count[''] #刪掉空白值
    count=sorted(count.items(), key=lambda item: item[1], reverse=True) #排序後 變成list
    i=0
    names=[]
    plt.figure(figsize=(16,6)) #畫長條圖 只取前十名
    for i in range(r):
        plt.bar(i,count[i][1],tick_label=count[i][0])
        names+=[count[i][0]]

    plt.xticks(range(0,r),names)
    plt.title(title)
    plt.savefig(pngname)
    plt.show()
    return count
