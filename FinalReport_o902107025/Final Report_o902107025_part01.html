def importomysql(k,job,company,salary,exp,indust,relist,toolist,jcontent,date,jlink):
    #
    dlink = pymysql.connect(
                host ="localhost",
                user="root",
                passwd="",
                db="job" #
                )
    #
    cmd = dlink.cursor()

    ## jobinfo 
    n=0
    
    #list 
    for j in job:
        c=company[n]
        s=str(salary[n])
        e=exp[n]
        i=indust[n]
        r=relist[n]
        t=toolist[n]
        jc=jcontent[n]
        d=date[n]
        l=jlink[n]

        if cmd.execute("select * from `104` where `jobtitle`= %s and `company`=%s",(j,c)) == 0: # 
            #
            pm= {
                "keyword":k,
                "jobtitle":j,
                "date":d,
                "company":c,
                "salary":s,
                "experiencetime":e,
                "industry":i,
                "requirement":r,
                "requiretool":t,
                "jobcontent":jc,
                "joblink":l
                }
            
            # 
            cmd.execute("INSERT INTO `104`(`keyword`,`jobtitle`,`date`,`company`,`salary`,`experiencetime`,`industry`,`requirement`,`requiretool`,`jobcontent`,`link`) VALUES(%(keyword)s,%(jobtitle)s,%(date)s,%(company)s,%(salary)s,%(experiencetime)s,%(industry)s,%(requirement)s,%(requiretool)s,%(jobcontent)s,%(joblink)s)",pm)
            dlink.commit #
            print('mysql', n)
            
        elif cmd.execute("select `id`,`keyword` from `104` where `jobtitle`= %s and `company`=%s",(j,c)) !=0: #
            k0=cmd.fetchall()[0] #
            Id=k0[0]     #id
            if k not in k0[1] : # 
                k1=k0[1]+' '+k
                cmd.execute("update `104` set `keyword`=%s where `jobtitle`= %s and `company`=%s",(k1,j,c))
                dlink.commit()
                print('mysql,id=%s,:%s,%s'%(Id,k,n))
            else: # 
                print('mysql:%s,id=%s,%s'%(k,Id,n))
        elif cmd.execute("select `id`,`keyword` from `104` where `jobtitle`= %s and `company`=%s",(j,c)) != 0:
            k0=cmd.fetchall()[0]
            Id=k0[0]
            print('mysql,id=%s,%s'%(Id,n))
        else:
            print('!!')
        n+=1

    dlink.close()
    print('mysql')

def counts(df,column,title,pngname,r):
    # 
    b=[]
    count={}
    for tt in df[column] :
        t=tt.split(' ')
        for i in t:
            if i not in b:
                count.update({i:1})
                b.append(i)
            else:
                count[i]+=1
    if '' in count:
        del count[''] #
    count=sorted(count.items(), key=lambda item: item[1], reverse=True) # list
    i=0
    names=[]
    plt.figure(figsize=(16,6)) # 
    for i in range(r):
        plt.bar(i,count[i][1],tick_label=count[i][0])
        names+=[count[i][0]]

    plt.xticks(range(0,r),names)
    plt.title(title)
    plt.savefig(pngname)
    plt.show()
    return count
